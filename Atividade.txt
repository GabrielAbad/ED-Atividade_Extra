void Power(int &iValor, const int iExpoente) {
    int iTmp = 1;
    for (int i = 0; i < iExpoente; i++) {
        iTmp = (iValor) * iTmp;
    }
    iValor = iTmp;
}

void Fact(int &iValor) {
    int iTmp = iValor;
    int iResultado = 1;
    for (int i = 1; i <= iValor; i++) {
        iResultado *= i;
    }
    iValor = iResultado;
}

float AritProg(float &fPrimeiro, float &fUltimo, float fTamanho) {
    return fTamanho * (fPrimeiro + fUltimo) / 2;
}

int CalcFib(int n) {
    if (n <= 1) {
        return 1;
    } else {
        return CalcFib(n - 1) + CalcFib(n - 2);
    }
}

float ContaDigitos(int &iDigito) {
    int iTmp = iDigito;
    int iContagemDigitos = 0;
    while (iTmp > 0) {
        iContagemDigitos++;
        iTmp = iTmp / 10;
    }
    return iContagemDigitos;
}
